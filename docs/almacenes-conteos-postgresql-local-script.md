BEGIN;

-- Script incremental para habilitar la agenda backend de conteos cíclicos de Almacenes.
-- Base analizada: C:\Users\exequ\OneDrive\Desktop\DBLocal.sql
-- Objetivo: agregar únicamente la tabla necesaria para publicar la planificación
-- y seguimiento básico de conteos desde backend, de forma segura e idempotente.

CREATE TABLE IF NOT EXISTS public.conteos_ciclicos (
    id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    deposito character varying(120) NOT NULL,
    zona character varying(120) NOT NULL,
    frecuencia character varying(60) NOT NULL,
    proximo_conteo date NOT NULL,
    estado character varying(30) NOT NULL DEFAULT 'programado',
    divergencia_pct numeric(9,2) NOT NULL DEFAULT 0,
    responsable character varying(150) NOT NULL DEFAULT 'Sin responsable',
    observacion character varying(500) NOT NULL DEFAULT 'Sin observación adicional.',
    next_step character varying(500) NOT NULL DEFAULT 'Sin próximo paso definido.',
    execution_note character varying(500) NOT NULL DEFAULT ''
);

ALTER TABLE public.conteos_ciclicos ADD COLUMN IF NOT EXISTS deposito character varying(120);
ALTER TABLE public.conteos_ciclicos ADD COLUMN IF NOT EXISTS zona character varying(120);
ALTER TABLE public.conteos_ciclicos ADD COLUMN IF NOT EXISTS frecuencia character varying(60);
ALTER TABLE public.conteos_ciclicos ADD COLUMN IF NOT EXISTS proximo_conteo date;
ALTER TABLE public.conteos_ciclicos ADD COLUMN IF NOT EXISTS estado character varying(30);
ALTER TABLE public.conteos_ciclicos ADD COLUMN IF NOT EXISTS divergencia_pct numeric(9,2);
ALTER TABLE public.conteos_ciclicos ADD COLUMN IF NOT EXISTS responsable character varying(150);
ALTER TABLE public.conteos_ciclicos ADD COLUMN IF NOT EXISTS observacion character varying(500);
ALTER TABLE public.conteos_ciclicos ADD COLUMN IF NOT EXISTS next_step character varying(500);
ALTER TABLE public.conteos_ciclicos ADD COLUMN IF NOT EXISTS execution_note character varying(500);

ALTER TABLE public.conteos_ciclicos ALTER COLUMN estado SET DEFAULT 'programado';
ALTER TABLE public.conteos_ciclicos ALTER COLUMN divergencia_pct SET DEFAULT 0;
ALTER TABLE public.conteos_ciclicos ALTER COLUMN responsable SET DEFAULT 'Sin responsable';
ALTER TABLE public.conteos_ciclicos ALTER COLUMN observacion SET DEFAULT 'Sin observación adicional.';
ALTER TABLE public.conteos_ciclicos ALTER COLUMN next_step SET DEFAULT 'Sin próximo paso definido.';
ALTER TABLE public.conteos_ciclicos ALTER COLUMN execution_note SET DEFAULT '';

UPDATE public.conteos_ciclicos
SET estado = COALESCE(NULLIF(TRIM(estado), ''), 'programado'),
    divergencia_pct = COALESCE(divergencia_pct, 0),
    responsable = COALESCE(NULLIF(TRIM(responsable), ''), 'Sin responsable'),
    observacion = COALESCE(NULLIF(TRIM(observacion), ''), 'Sin observación adicional.'),
    next_step = COALESCE(NULLIF(TRIM(next_step), ''), 'Sin próximo paso definido.'),
    execution_note = COALESCE(execution_note, '')
WHERE estado IS NULL
   OR TRIM(estado) = ''
   OR divergencia_pct IS NULL
   OR responsable IS NULL
   OR TRIM(responsable) = ''
   OR observacion IS NULL
   OR TRIM(observacion) = ''
   OR next_step IS NULL
   OR TRIM(next_step) = ''
   OR execution_note IS NULL;

ALTER TABLE public.conteos_ciclicos ALTER COLUMN deposito SET NOT NULL;
ALTER TABLE public.conteos_ciclicos ALTER COLUMN zona SET NOT NULL;
ALTER TABLE public.conteos_ciclicos ALTER COLUMN frecuencia SET NOT NULL;
ALTER TABLE public.conteos_ciclicos ALTER COLUMN proximo_conteo SET NOT NULL;
ALTER TABLE public.conteos_ciclicos ALTER COLUMN estado SET NOT NULL;
ALTER TABLE public.conteos_ciclicos ALTER COLUMN divergencia_pct SET NOT NULL;
ALTER TABLE public.conteos_ciclicos ALTER COLUMN responsable SET NOT NULL;
ALTER TABLE public.conteos_ciclicos ALTER COLUMN observacion SET NOT NULL;
ALTER TABLE public.conteos_ciclicos ALTER COLUMN next_step SET NOT NULL;
ALTER TABLE public.conteos_ciclicos ALTER COLUMN execution_note SET NOT NULL;

DO $$
BEGIN
    IF NOT EXISTS (
        SELECT 1
        FROM pg_constraint
        WHERE conname = 'ck_conteos_ciclicos_estado'
    ) THEN
        ALTER TABLE public.conteos_ciclicos
            ADD CONSTRAINT ck_conteos_ciclicos_estado
            CHECK (estado IN ('programado', 'en-ejecucion', 'observado'));
    END IF;
END $$;

DO $$
BEGIN
    IF NOT EXISTS (
        SELECT 1
        FROM pg_constraint
        WHERE conname = 'ck_conteos_ciclicos_divergencia'
    ) THEN
        ALTER TABLE public.conteos_ciclicos
            ADD CONSTRAINT ck_conteos_ciclicos_divergencia
            CHECK (divergencia_pct >= 0);
    END IF;
END $$;

CREATE INDEX IF NOT EXISTS ix_conteos_ciclicos_estado
    ON public.conteos_ciclicos (estado);

CREATE INDEX IF NOT EXISTS ix_conteos_ciclicos_proximo_conteo
    ON public.conteos_ciclicos (proximo_conteo);

COMMIT;
