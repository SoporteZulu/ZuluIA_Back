# SQL document: `zuluia_back_terceros_medios_contacto_upgrade`

Fix operativo para PostgreSQL local cuando `GET /api/terceros/{id}` o `GET /api/terceros/{id}/medios-contacto` falla con `42P01: no existe la relación «PER_MEDIOCONTACTO»`.

## Contexto

- El backend tolera temporalmente la ausencia de `PER_MEDIOCONTACTO` y devuelve `mediosContacto: []` para no romper el detalle.
- Para recuperar la funcionalidad completa de medios de contacto, crear la tabla faltante en `localhost:5432`.

## SQL a ejecutar manualmente

```sql
BEGIN;

CREATE TABLE IF NOT EXISTS public.per_tipomediocontacto
(
    tmc_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    tmc_codigo varchar(20) NOT NULL,
    tmc_descripcion varchar(100) NOT NULL,
    tmc_activo boolean NOT NULL DEFAULT true,
    created_at timestamp with time zone NOT NULL DEFAULT now(),
    updated_at timestamp with time zone NOT NULL DEFAULT now(),
    deleted_at timestamp with time zone NULL,
    created_by bigint NULL,
    updated_by bigint NULL
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_per_tipomediocontacto_codigo
    ON public.per_tipomediocontacto (tmc_codigo);

CREATE TABLE IF NOT EXISTS public.per_mediocontacto
(
    mcon_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    per_id bigint NOT NULL,
    tmc_id bigint NULL,
    mcon_valor varchar(200) NOT NULL,
    mcon_orden integer NOT NULL DEFAULT 0,
    mcon_defecto boolean NOT NULL DEFAULT false,
    mcon_observacion varchar(500) NULL
);

CREATE INDEX IF NOT EXISTS ix_per_mediocontacto_per_id
    ON public.per_mediocontacto (per_id);

CREATE INDEX IF NOT EXISTS ix_per_mediocontacto_tmc_id
    ON public.per_mediocontacto (tmc_id);

DO $$
BEGIN
    IF NOT EXISTS (
        SELECT 1
        FROM pg_constraint
        WHERE conname = 'fk_per_mediocontacto_terceros_per_id') THEN
        ALTER TABLE public.per_mediocontacto
            ADD CONSTRAINT fk_per_mediocontacto_terceros_per_id
            FOREIGN KEY (per_id) REFERENCES public.terceros (id) ON DELETE CASCADE;
    END IF;
END $$;

DO $$
BEGIN
    IF NOT EXISTS (
        SELECT 1
        FROM pg_constraint
        WHERE conname = 'fk_per_mediocontacto_tipos_tmc_id') THEN
        ALTER TABLE public.per_mediocontacto
            ADD CONSTRAINT fk_per_mediocontacto_tipos_tmc_id
            FOREIGN KEY (tmc_id) REFERENCES public.per_tipomediocontacto (tmc_id) ON DELETE SET NULL;
    END IF;
END $$;

INSERT INTO public.per_tipomediocontacto (tmc_codigo, tmc_descripcion, tmc_activo)
VALUES
    ('TEL', 'Teléfono', true),
    ('CEL', 'Celular', true),
    ('EMAIL', 'Correo electrónico', true),
    ('WEB', 'Sitio web', true),
    ('OTRO', 'Otro', true)
ON CONFLICT (tmc_codigo) DO NOTHING;

COMMIT;
```
